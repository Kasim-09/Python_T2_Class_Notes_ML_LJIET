import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import PolynomialFeatures
from sklearn.neighbors import KNeighborsClassifier
from sklearn.tree import DecisionTreeClassifier
from sklearn.svm import SVC
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import r2_score,accuracy_score,confusion_matrix,classification_report

f = pd.read_csv("data.csv")
df = pd.DataFrame(f)
x = df[["in"]]
y = df["out"]
x = pd.get_dummies(x) #if string column in independent data
x_train,x_test,y_train,y_test = train_test_split(x,y,test_size=0.2,random_state=42)

# Linear Regression
model = LinearRegression()
ans = model.fit(x_train,y_train)
pre = model.predict(x_test)
acc = r2_score(y_test,pre)
print(acc)

# Polynomial Regression
poly = PolynomialFeatures(degree=2)
x_poly_train=poly.fit_transform(x_train)
x_poly_test=poly.transform(x_test)

model = LinearRegression()
ans = model.fit(x_poly_train,y_train)
pre = model.predict(x_poly_test)
acc = r2_score(y_test,pre)
print(acc)

#KNN
model = KNeighborsClassifier(n_neighbors=3)
ans = model.fit(x_train,y_train)
pre = model.predict(x_test)
acc = accuracy_score(y_test,pre)
print(acc)

#DecisionTreeClassifier
model = DecisionTreeClassifier(criterion="entropy",max_depth=3,random_state=42)
ans = model.fit(x_train,y_train)
pre = model.predict(x_test)
acc = accuracy_score(y_test,pre)
print(acc)

#SVC
model = SVC(kernel="rbf",max_depth=3,random_state=42)
ans = model.fit(x_train,y_train)
pre = model.predict(x_test)
acc = accuracy_score(y_test,pre)
print(acc)

#RandomForestClassifier
model = RandomForestClassifier(n_estimators=3,criterion="entropy",max_depth=3,random_state=42)
ans = model.fit(x_train,y_train)
pre = model.predict(x_test)
acc = accuracy_score(y_test,pre)
print(acc)
